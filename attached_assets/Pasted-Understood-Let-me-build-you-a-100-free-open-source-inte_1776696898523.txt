### 1. PK-Sim/MoBi — The Free PBPK Engine

**What you get for free:**
- Full animal physiology libraries (mouse, rat, dog, monkey, human)
- Whole-body PBPK with organ-specific distribution
- Permeability-limited brain model for CNS exposure
- Command-line batch processing
- R integration via `ospsuite` package


**Batch Automation Script (R):**
```r
library(ospsuite)
library(tlf)

# Load species template (free with software)
rat_model <- loadSimulation("rat_pbpk_template.pkml")

# Function: In Vitro -> Animal PK
predict_animal_pk <- function(compound_data, species = "rat") {
  
  # Load species-specific model
  model_file <- paste0(species, "_template.pkml")
  sim <- loadSimulation(model_file)
  
  # Set compound parameters from your In Vitro data
  setParameterValuesByPath(
    "Compound|Lipophilicity", 
    compound_data$logP, 
    simulation = sim
  )
  setParameterValuesByPath(
    "Compound|Solubility", 
    compound_data$solubility, 
    simulation = sim
  )
  setParameterValuesByPath(
    "Compound|Permeability", 
    compound_data$peff, 
    simulation = sim
  )
  setParameterValuesByPath(
    "Compound|IntrinsicClearance", 
    compound_data$clint, 
    simulation = sim
  )
  setParameterValuesByPath(
    "Compound|FractionUnbound", 
    1 - compound_data$fup, 
    simulation = sim
  )
  
  # Run simulation
  results <- runSimulation(sim)
  
  # Extract PK parameters
  pk <- list(
    auc = calculateAUC(results, "Organism|PeripheralVenousBlood|Plasma|Concentration"),
    cmax = calculateCmax(results, "Organism|PeripheralVenousBlood|Plasma|Concentration"),
    tmax = calculateTmax(results, "Organism|PeripheralVenousBlood|Plasma|Concentration"),
    clearance = calculateClearance(results),
    vd = calculateVolumeOfDistribution(results),
    brain_cmax = calculateCmax(results, "Organism|Brain|Intracellular|Concentration")
  )
  
  return(pk)
}

# Batch process multiple compounds
compounds <- read.csv("your_in_vitro_data.csv")
results <- lapply(compounds, predict_animal_pk, species = "rat")
```

```python
# API access (free)
from chembl_webresource_client.new_client import new_client

def get_animal_pk_data(chembl_id):
    client = new_client()
    # Filter for assay type = "ADME" and organism = "Rattus norvegicus" or "Canis lupus familiaris"
    assays = client.assay.filter(
        target_chembl_id=chembl_id,
        assay_type="ADME",
        organism__in=["Rattus norvegicus", "Canis lupus familiaris"]
    )
    return assays
```

## Adding Population Variability (Free)

Since Simcyp's virtual populations are paid, use **nlmixr2** (free R package):

```r
library(nlmixr2)
library(rxode2)

# Define population PK model from PK-Sim outputs
pk_model <- function() {
  ini({
    tcl <- log(0.5)    # Typical clearance
    tv <- log(50)      # Typical volume
    eta.cl ~ 0.09      # IIV on clearance
    eta.v ~ 0.09       # IIV on volume
    add.err <- 0.1     # Residual error
  })
  model({
    cl <- exp(tcl + eta.cl)
    v <- exp(tv + eta.v)
    ke <- cl/v
    d/dt(central) <- -ke * central
    cp <- central/v + add.err
  })
}

# Simulate 100 virtual animals
sim <- rxSolve(pk_model, nSub = 100, events = dosing_regimen)
```
