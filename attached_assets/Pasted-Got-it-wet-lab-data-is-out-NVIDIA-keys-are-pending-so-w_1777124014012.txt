// src/types/quality.ts

export type QualityTier = 'experimental' | 'enriched' | 'heuristic' | 'inferred';

export interface SourceQuality {
  score: number;        // 1.0, 0.7, 0.3, 0.1
  tier: QualityTier;
  source: string;       // "ChEMBL", "RDKit heuristic", "BioNeMo", etc.
  provenance?: string;  // URL, assay ID, or file reference
  timestamp: string;    // ISO 8601
}

export interface QualityScoredValue<T> {
  value: T;
  quality: SourceQuality;
}

export interface ChEMBLQualityPayload {
  nMeasurements: number;
  range_nM: [number, number];
  geometricMean_nM: number;
  species: string[];
  assayTypes: Record<string, number>;
  assayFormats: Record<string, number>;
  confidence: 'high' | 'medium' | 'low';
}

export interface EnrichedChEMBLTarget {
  uniprotId: string;
  targetName: string;
  bestValue_nM: number;
  quality: SourceQuality & { payload: ChEMBLQualityPayload };
}

## Cascade Gate Rules (cascadeGate.ts)

```typescript
export interface CascadeDecision {
  canAutoAdvance: boolean;
  reason: string;
  blockedBy: string[];  // which outputs failed the gate
  requiresOverride: boolean;
}

export function evaluateCascadeGate(
  outputs: Record<string, QualityScoredValue<any>>,
  goNoGoDecision: 'GO' | 'WATCH' | 'NO-GO'
): CascadeDecision {
  
  const SAFETY_CRITICAL_KEYS = [
    'hERG_IC50', 'hepatotoxicityRisk', 'genotoxicityAlert', 'cypInhibition'
  ];
  
  if (goNoGoDecision === 'NO-GO') {
    return {
      canAutoAdvance: false,
      reason: 'NO-GO compounds require manual review regardless of data quality',
      blockedBy: ['goNoGo'],
      requiresOverride: true
    };
  }
  
  const blockedBy: string[] = [];
  
  for (const key of SAFETY_CRITICAL_KEYS) {
    const output = outputs[key];
    if (!output || output.quality.score < 0.6) {
      blockedBy.push(key);
    }
  }
  
  const nonSafetyKeys = Object.keys(outputs).filter(k => !SAFETY_CRITICAL_KEYS.includes(k));
  const nonSafetyScores = nonSafetyKeys.map(k => outputs[k]?.quality.score ?? 0);
  const avgNonSafety = nonSafetyScores.length > 0 
    ? nonSafetyScores.reduce((a, b) => a + b, 0) / nonSafetyScores.length 
    : 0;
  
  if (avgNonSafety < 0.6) {
    blockedBy.push('average_non_safety_quality');
  }
  
  return {
    canAutoAdvance: blockedBy.length === 0,
    reason: blockedBy.length === 0 
      ? 'All quality thresholds met' 
      : `Blocked by: ${blockedBy.join(', ')}`,
    blockedBy,
    requiresOverride: blockedBy.length > 0
  };
}
```

## ChEMBL v2 API Contract

**Route:** `POST /api/chembl/similar-targets/v2`

**Request:**
```json
{
  "smiles": "CC(C)Cc1ccc(C(C)C(=O)O)cc1",
  "similarityThreshold": 0.7,
  "maxResults": 5
}
```

**Response:**
```json
{
  "querySmiles": "CC(C)Cc1ccc(C(C)C(=O)O)cc1",
  "targets": [
    {
      "uniprotId": "P00533",
      "targetName": "EGFR",
      "bestValue_nM": 1.2,
      "bestActivityType": "IC50",
      "bestPChembl": 8.92,
      "bestSimilarity": 0.87,
      "quality": {
        "score": 1.0,
        "tier": "experimental",
        "source": "ChEMBL",
        "provenance": "https://www.ebi.ac.uk/chembl/target/UNIPROT_P00533",
        "timestamp": "2026-04-25T17:32:00Z",
        "payload": {
          "nMeasurements": 8,
          "range_nM": [0.4, 18.0],
          "geometricMean_nM": 3.4,
          "species": ["human", "human", "mouse"],
          "assayTypes": {"IC50": 5, "Ki": 2, "Kd": 1},
          "assayFormats": {"biochemical": 6, "cell-based": 2},
          "confidence": "medium"
        }
      }
    }
  ],
  "fallbackToV1": false
}
```

**UI rendering rule:**
- `confidence === "high"` AND `species` all human AND one dominant `assayType`: render point estimate normally
- `confidence === "medium"`: render range as primary, point estimate secondary
- `confidence === "low"` OR `nMeasurements < 3`: render **"Insufficient data"** with link to raw ChEMBL records, suppress point estimate in cascade calculations
